157  inline void PID<T>::update(T target, T now_val, T dt)
158  {
159  diff[0] = target - now_val; // 最新の偏差
160  integral += (diff[0] + diff[1]) * (dt / 2.0); // 積分
161 
162  switch (_param.mode)

164  case Mode::pPID:

175  diff[2] = diff[1];
176  diff[1] = diff[0];
